139 template <class T>
140 T* MappedFile::Allocate(size_t count) {
141  if (!IsOpen())
142  return NULL;
143 
144  size_t used_space = RIME_ALIGNED(size_, T);
145  size_t required_space = sizeof(T) * count;
146  size_t file_size = capacity();
147  if (used_space + required_space > file_size) {
148  // not enough space; grow the file
149  size_t new_size = (std::max)(used_space + required_space, file_size * 2);
150  if(!Resize(new_size) || !OpenReadWrite())
151  return NULL;
152  }
153  T* ptr = reinterpret_cast<T*>(address() + used_space);
154  std::memset(ptr, 0, required_space);
155  size_ = used_space + required_space;
156  return ptr;
157 }
